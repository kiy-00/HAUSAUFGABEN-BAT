int is_youxiao(char ch)
{
	if (ch >= '0' && ch <= '9')
		return ch - '0';
	else if (ch >= 'a' && ch <= 'z')
		return ch - 'a' + 10;
	else if (ch >= 'A' && ch <= 'Z')
		return ch - 'A' + 10;
	else
		return -1;
}

/* è½¬åŒ–åè¿›åˆ¶ */
int transform_ten(char* p,int len)
{
	int val = 0;
	for (int i = 0; i < len; i++)
	{
		int temp = is_youxiao(p[i]);
		val = 16 * val + temp;
	}
	return val;
}
